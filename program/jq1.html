	class student
{
   int rollno;
   String name;
   int marks;
   student()
  {
    rollno=1111;
    name="Varun";
    marks=470;
  }
  student(int rno,String n,int m)
   {
     rollno=rno;
     name=n;
     marks=m;
    }
  float compute_percentage()
  {
   float f;
   f=marks*100/625;
   return(f);
  }
  void print()
  {
     System.out.println("Rollno-"+rollno);
     System.out.println("Name-"+name);
     System.out.println("Marks-"+marks);
  }
}
class studdemo
{
  public static void main(String arg[])
  {
     student s1=new student();
     s1.print();
     System.out.println("Percentage is "+s1.compute_percentage());
     System.out.println();
     student s2=new student(2222,"Demo",620);
     s2.print();
     System.out.println("Percentage is "+s2.compute_percentage());
  }
}
